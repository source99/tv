FROM  `tvseries` ,  `tvseasons` ,  `tvepisodes` 
WHERE  `tvseries`.`id` =  `tvseasons`.`seriesid` 
AND  `tvseasons`.`id` =  `tvepisodes`.`seasonid` 
AND  `tvseries`.`SeriesName` LIKE  '%big bang theory%'
ORDER BY `tvepisodes`.`firstaired` DESC
